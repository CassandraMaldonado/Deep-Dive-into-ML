def extract_zip(zip_path, extract_to):
    # Create extraction directory if it doesn't exist
    if not os.path.exists(extract_to):
        os.makedirs(extract_to)
    
    # Extract the zip file
    with zipfile.ZipFile(zip_path, 'r') as zip_ref:
        zip_ref.extractall(extract_to)
    
    print(f"Extracted {zip_path} to {extract_to}")
    return extract_to

def organize_test_files(test_dir):
    if not os.path.exists(test_dir):
        print(f"Test directory {test_dir} does not exist")
        return False
    
    organized_dir = test_dir + "_organized"
    if os.path.exists(organized_dir):
        shutil.rmtree(organized_dir)
    os.makedirs(organized_dir)
    
    # Class directories
    for i in range(1, 5):
        os.makedirs(os.path.join(organized_dir, f"category {i}"))
    
    # Map file prefixes to categories
    prefix_mapping = {
        "10": "category 1",  
        "40": "category 2",  
        "60": "category 3",  
        "C0": "category 4" 
    }
    
    files_moved = 0
    for filename in os.listdir(test_dir):
        if filename.endswith(".png"):
            prefix = filename[:2]
            
            # Determining target category based on prefix
            target_category = None
            for key, category in prefix_mapping.items():
                if prefix.startswith(key):
                    target_category = category
                    break
            
            if target_category is None:
                print(f"Could not determine category for {filename}, using category 1 as default")
                target_category = "category 1"
            
            # Copy the file
            source_path = os.path.join(test_dir, filename)
            target_path = os.path.join(organized_dir, target_category, filename)
            shutil.copy2(source_path, target_path)
            files_moved += 1
    
    print(f"Organized {files_moved} test files into class directories")
    return organized_dir

2. Initial Classifier Build:¶

Now use keras to build an initial image classifier with the following specifications.

Note: If you get lost, there is great documentation online and homework 7 included details on many of the layers used here.

  • Create an instance of Sequential called "classifier"
  • Add a Conv2D layer with the following parameters:
    • filters = 32
    • kernel_size = (3,3)
    • input_shape = image shape found in part 1
    • activation = relu
  • Add a MaxPooling2D layer where pool_size = (2,2)
  • Add another Conv2D layer:
    • filters = 64
    • kernel_size = (3,3)
    • activation = relu
  • Add a MaxPooling2D layer where pool_size = (2,2)
  • Add a Flatten layer
  • Add a Dense layer
    • units = 128
    • activation = relu
  • Add a final Dense layer (this will output our probabilities):
    • units = # of classes
    • activation = softmax
  • Compile with the following:
    • optimize = adam
    • loss = categorical cross entropy
    • metric = accuracy

model = keras.Sequential([
    # First Convolutional Layer
    keras.layers.Conv2D(32, (3, 3), input_shape=(64, 64, 3), activation='relu'),
    keras.layers.MaxPooling2D(pool_size=(2, 2)),
    
    # Second Convolutional Layer
    keras.layers.Conv2D(64, (3, 3), activation='relu'),
    keras.layers.MaxPooling2D(pool_size=(2, 2)),

    keras.layers.Conv2D(128, (3, 3), activation='relu'),
    keras.layers.MaxPooling2D(pool_size=(2, 2)),
    
    # Flattening
    keras.layers.Flatten(),
    
    # Full Connection
    keras.layers.Dense(128, activation='relu'),
    keras.layers.Dropout(0.5),  # Add dropout to prevent overfitting
    keras.layers.Dense(4, activation='softmax')  # 4 output nodes for 4 categories
])

def predict_image(image_path, model):
    test_image = image.load_img(image_path, target_size=(64, 64))
    test_image = image.img_to_array(test_image)
    test_image = np.expand_dims(test_image, axis=0)
    test_image = test_image / 255.0  # Rescale
    
    prediction = model.predict(test_image)
    class_indices = {v: k for k, v in training_set.class_indices.items()}
    predicted_class = class_indices[np.argmax(prediction)]
    confidence = np.max(prediction)
    
    return predicted_class, confidence

# Training with different parameters
def train_model_with_params(steps_per_epoch, epochs):
    print(f"\nTraining with steps_per_epoch={steps_per_epoch}, epochs={epochs}")
    
    # New model for each run
    new_model = keras.Sequential([
        keras.layers.Conv2D(32, (3, 3), input_shape=(64, 64, 3), activation='relu'),
        keras.layers.MaxPooling2D(pool_size=(2, 2)),
        keras.layers.Conv2D(64, (3, 3), activation='relu'),
        keras.layers.MaxPooling2D(pool_size=(2, 2)),
        keras.layers.Conv2D(128, (3, 3), activation='relu'),
        keras.layers.MaxPooling2D(pool_size=(2, 2)),
        keras.layers.Flatten(),
        keras.layers.Dense(128, activation='relu'),
        keras.layers.Dropout(0.5),
        keras.layers.Dense(4, activation='softmax')
    ])
    
    new_model.compile(optimizer='adam',
                     loss='categorical_crossentropy',
                     metrics=['accuracy'])
    
    # Training
    history = new_model.fit(
        training_set,
        steps_per_epoch=steps_per_epoch,
        epochs=epochs,
        validation_data=test_set,
        validation_steps=test_set.samples // test_set.batch_size
    )
    
    # Evaluate the model
    test_loss, test_accuracy = new_model.evaluate(test_set)
    
    # Save the model with parameters in the filename
    model_filename = f'model_steps{steps_per_epoch}_epochs{epochs}.h5'
    new_model.save(model_filename)
    
    return {
        'steps_per_epoch': steps_per_epoch,
        'epochs': epochs,
        'test_accuracy': test_accuracy,
        'test_loss': test_loss,
        'train_accuracy': history.history['accuracy'][-1],
        'train_loss': history.history['loss'][-1],
        'val_accuracy': history.history['val_accuracy'][-1] if 'val_accuracy' in history.history else None,
        'val_loss': history.history['val_loss'][-1] if 'val_loss' in history.history else None,
        'history': history
    }

Conceptual Questions:¶

  1. Discuss the effect of the following on accuracy and loss (train & test):
  • Increasing the steps_per_epoch.
  • Increasing the number of epochs.

Increasing steps per epoch: When we increase steps perepoch, we're basically telling the model to look at more batches of training data before finishing one complete epoch. I noticed that the model sees more training examples in each epoch, so it gets more chances to learn. As we increaste the steps the training takes longer since we're processing more data per epoch, also there can be more fluctuations in the accuracy and loss during training. Is important to know that if steps per epoch is too high, we might end up reusing data in a single epoch which isn't ideal. In the test accuracy, it doesn't directly affect it since we test on all test data anyway, but a better trained model should perform better on the test set too.

Increasing the number of epochs: This means how many complete passes we make through the training data. When I increased epochs the model initially got better as it had more time to learn, but after a certain point, the training accuracy kept improving but the test accuracy started to get worse. This showed me that too many epochs can cause overfitting, since the model just memorizes the training data. I found it's best to stop training when validation loss starts increasing even if training loss is still decreasing.

I got the best results with a moderate number of epochs 5-8 paired with enough steps per epoch to cover most of my training data. Going too high with either parameter led to diminishing returns and potential overfitting.

  1. Name two uses of zero padding in CNN.

1. Keeping the size of the feature maps the same: Without padding, each convolutional layer makes the image smaller, which is a problem if we have a deep network. With zero padding, we can maintain the same height and width after convolution. This was super helpful in my model since I could control exactly how much downsampling happened through pooling layers rather than losing dimensions unintentionally.

2. Not losing edge information: In class we learned that without padding, the pixels at the edges get used way less in convolutions than center pixels. With zero padding, the edges get processed just as much as the center. If some of the key features are near the edges of images, padding can help the model detect those patterns.
